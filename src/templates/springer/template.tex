% Springer-style monograph template (XeLaTeX)
% --------------------------------------------------------------------
% This is a production-ready, minimal Springer-like template geared for
% auto-generation. It expects the exporter to:
%   - replace placeholders %%TITLE%%, %%SUBTITLE%%, %%AUTHORS%%, %%DATE%%
%   - write chapter .tex files into ./chapters/*.tex (optional)
%   - write figures under ./figures and tables under ./tables (optional)
%   - optionally drop a references.bib file for biblatex
%
% Engine: xelatex
% --------------------------------------------------------------------

\documentclass[graybox,envcountchap,sectrefs]{svmono}

% Project preamble (packages, fonts, listings, links)
\input{preamble.tex}

% ---------------------------- METADATA --------------------------------
\title{%%TITLE%%}
\subtitle{%%SUBTITLE%%}
\author{%%AUTHORS%%}
\date{%%DATE%%}

% If your build writes a PDF/A compliant metadata block, keep the hyperref
% setup in preamble.tex and avoid redefining pdfauthor/pdftitle here.

% ----------------------------- DOCUMENT -------------------------------
\begin{document}
\frontmatter
\maketitle
\tableofcontents

% --------- Optional dedication, preface, acknowledgments ---------------
\chapter*{Preface}
This book was generated with the Workshop Builder pipeline. It can be edited like any other \LaTeX{} project. Replace this preface with your own introduction, scope and audience notes.

\mainmatter

% --------------------------- CHAPTER INPUTS ----------------------------
% The exporter may either:
%  (A) write one big chapters/000-main.tex and include it,
%  (B) write many chapters/*.tex and \include them, or
%  (C) inline content directly here (less ideal for large works).
%
% Strategy A (single assembled file) â€” default for auto-exporters:
\IfFileExists{chapters/000-main.tex}{\input{chapters/000-main.tex}}{}

% Strategy B (per-chapter files):
\IfFileExists{chapters/01-introduction.tex}{\include{chapters/01-introduction}}{}
\IfFileExists{chapters/02-architecture.tex}{\include{chapters/02-architecture}}{}
\IfFileExists{chapters/03-agents.tex}{\include{chapters/03-agents}}{}
\IfFileExists{chapters/04-rag.tex}{\include{chapters/04-rag}}{}
\IfFileExists{chapters/05-labs.tex}{\include{chapters/05-labs}}{}
% (Your exporter can append more \include lines as needed.)

% --------------------------- APPENDICES --------------------------------
\appendix
\IfFileExists{chapters/A-appendix.tex}{\include{chapters/A-appendix}}{}

% --------------------------- BIBLIOGRAPHY ------------------------------
% Use biblatex (configured in preamble.tex). If references.bib exists,
% print the bibliography; otherwise this section will be omitted.
\IfFileExists{references.bib}{
  \printbibliography[heading=bibintoc,title={References}]
}{}

\backmatter
\end{document}
